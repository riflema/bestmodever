	usage = game
	selection_behavior = weighted_random
	fallback = no_hands
	
	no_hands = {
        dna_modifiers = {
            accessory = {
                mode = add
                gene = special_accessories_hands
                template = no_hands
                range = { 0 1 } # For the randomness to work correctly
            }
        } 
    }
	
	western_high_nobility_gloves = {
		dna_modifiers = {
			accessory = {
				mode = add
				gene = special_accessories_hands
				template = western_high_nobility_gloves
				range = { 0 1 } # For the randomness to work correctly
			}
		}
		weight = {
			base = 0
			modifier = {
				add = 100
				
				cfp_era_high_medieval_trigger = yes
				
				OR = {
					AND = {
						OR = {
							highest_held_title_tier >= tier_duchy
							AND = {
								has_government = republic_government
								primary_title.tier > 1
							}
							AND = { is_ruler = yes has_government = theocracy_government }
						}
						NOT = { AND = { is_ruler = yes has_government = mercenary_government  } } # Blocked for mercenaries
						portrait_western_clothing_trigger = yes
					}
					AND = {
						exists = primary_spouse
						OR = {
							is_ruler = no
							highest_held_title_tier = tier_barony
						}
						primary_spouse = {
							OR = {
								highest_held_title_tier >= tier_duchy
								AND = {
									has_government = republic_government
									primary_title.tier > 1
								}
								AND = { is_ruler = yes has_government = theocracy_government }
							}
							portrait_western_clothing_trigger = yes
						}
					}
				}
			}
		}
	}
