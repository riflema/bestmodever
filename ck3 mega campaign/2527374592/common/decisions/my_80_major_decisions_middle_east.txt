#Create Turkish culture
embrace_turkish_culture_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_realm.dds"
	major = yes

	ai_check_interval = 120

	desc = embrace_turkish_culture_decision_desc

	is_shown = {
		has_culture = culture:turkish
		capital_province = {
			geographical_region = world_asia_minor
		}
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:embrace_turkish_culture_decision
			}
		}
	}

	is_valid = {
		prestige_level >= 3
		custom_description = {
			text = capital_in_asia_minor_region_tooltip
			capital_province = {
				geographical_region = world_asia_minor
			}
		}
		has_title = title:k_rum
	}

	is_valid_showing_failures_only = {
		is_imprisoned = no
		is_available_adult = yes
	}

	effect = {
		add_prestige = 150
		set_culture = culture:ottoman_turkish
		every_vassal_or_below = {
			custom = embrace_turkish_culture_vassals_custom
			limit = {
				culture = culture:turkish
				capital_province = {
					geographical_region = world_asia_minor
				}
			}
			set_culture = culture:ottoman_turkish
			every_courtier = {
				limit = {
					culture = culture:turkish
				}
				set_culture = culture:ottoman_turkish
			}
		}
		every_courtier_or_guest = {
			custom = embrace_turkish_culture_courtiers_custom
			limit = {
				culture = culture:turkish
			}
			set_culture = culture:ottoman_turkish
		}
		every_county = {
			custom = embrace_turkish_culture_counties_custom
			limit = {
				culture = culture:turkish
				title_province = {	
					geographical_region = world_asia_minor
				}
			}
			set_county_culture = culture:ottoman_turkish
		}
		culture:ottoman_turkish = {
			get_all_innovations_from = culture:turkish
			add_random_innovation = culture_group_military
			add_random_innovation = culture_group_military
			add_random_innovation = culture_group_military
			add_random_innovation = culture_group_civic
			add_random_innovation = culture_group_civic
			add_random_innovation = culture_group_civic
			add_random_innovation = culture_group_civic
		}
	}
		
	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}
