add_namespace = sic


# Secure the German Alliance event for Germany
country_event = {
	id = sic.1
	title = "Sicily wants a alliance"
	desc = sic.1.d
	picture = GFX_report_event_eng_mosley_mussolini

	is_triggered_only = yes
	
	# Accept
	option = {
		name = "A great idea"
		FROM = {
			if = {
				limit = {
					is_in_faction = no
				}
				create_faction = "The Anti Italian League"
			}
			add_to_faction = ROOT
			country_event = { id = sic.2 hours = 6 }
			hidden_effect = {
				add_ai_strategy = {
					type = alliance
					id = ROOT
					value = 200
				}
				add_ai_strategy = {
					type = befriend
					id = ROOT
					value = 200
				}
				add_ai_strategy = {
					type = antagonize
					id = ROOT
					value = -200
				}
			}
		}
		hidden_effect = {
			add_ai_strategy = {
				type = alliance
				id = FROM
				value = 200
			}
			add_ai_strategy = {
				type = befriend
				id = FROM
				value = 200
			}
			add_ai_strategy = {
				type = antagonize
				id = FROM
				value = -200
			}
		}
	}

	# Reject
	option = {
		name = "We dont need them"
		ai_chance = {
			factor = 0
		}
		FROM = { country_event = { id = sic.3 hours = 6 } }
	}
}
